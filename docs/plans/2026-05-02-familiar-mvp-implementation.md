# familiar MVP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a working v0.1 of `familiar` — a pixel-art pet companion for Claude Code CLI with statusline rendering, `/pet` command, affection system, memory backend, and proactive speech via hooks.

**Architecture:** TypeScript/Node.js npm package. The package exposes (a) a statusline binary called by Claude Code on every refresh, (b) slash command handlers registered as Claude Code custom commands, and (c) hook-driven listeners that react to Claude Code lifecycle events. State lives in `~/.familiar/state.json`. Pet personality is generated by calling the user's own Claude account via the `@anthropic-ai/sdk`.

**Tech Stack:**
- Node.js 18+ / TypeScript 5
- Build: `tsup` (zero-config bundler)
- Test: `vitest`
- Lint/format: `biome`
- Anthropic SDK: `@anthropic-ai/sdk`
- CLI prompts: `@inquirer/prompts`
- Distribution: npm (with `bin` entries for `familiar` and `familiar-statusline`)

**Hard prerequisite:** Claude Code CLI must be installed (`claude --version` succeeds). The installer fails fast and gives a friendly link if not.

import type { Sprite, RGB } from "../pets/types.js";

const RESET = "\x1b[0m";
const fg = ([r, g, b]: RGB) => `\x1b[38;2;${r};${g};${b}m`;
const bg = ([r, g, b]: RGB) => `\x1b[48;2;${r};${g};${b}m`;

export function renderSprite(sprite: Sprite): string {
  const { pixels, palette } = sprite;
  const rows = pixels.length;
  let out = "";

  for (let y = 0; y < rows - 1; y += 2) {
    const top = pixels[y];
    const bot = pixels[y + 1];
    for (let x = 0; x < top.length; x++) {
      const t = top[x];
      const b = bot[x];
      if (t === 0 && b === 0) {
        out += " ";
      } else if (t === 0) {
        const c = palette[b];
        out += `${bg(c)}${fg([0, 0, 0])}▄${RESET}`;
      } else if (b === 0) {
        const c = palette[t];
        out += `${fg(c)}▀${RESET}`;
      } else {
        const tc = palette[t];
        const bc = palette[b];
        out += `${bg(bc)}${fg(tc)}▀${RESET}`;
      }
    }
    out += "\n";
  }
  return out;
}
```

import { existsSync, mkdirSync, readFileSync, writeFileSync } from "node:fs";
import { dirname } from "node:path";
import { familiarHome, stateFile } from "../util/paths.js";
import type { PetState, Species } from "./types.js";

export function loadState(): PetState | null {
  const path = stateFile();
  if (!existsSync(path)) return null;
  return JSON.parse(readFileSync(path, "utf8")) as PetState;
}

export function saveState(state: PetState): void {
  const path = stateFile();
  if (!existsSync(familiarHome())) {
    mkdirSync(familiarHome(), { recursive: true });
  }
  writeFileSync(path, JSON.stringify(state, null, 2));
}

export function defaultState(opts: { species: Species; name: string }): PetState {
  const now = new Date().toISOString();
  return {
    schemaVersion: 1,
    species: opts.species,

    mood: "neutral",
    createdAt: now,
    lastInteractionAt: now,

export function applyDelta(current: number, delta: number): number {

];

function parsePythonArray(src, varName, symbols) {
  // Find: const varName = [ ... ] block (Python style: varName = [ ... ])
  const re = new RegExp(`${varName}\\s*=\\s*\\[([\\s\\S]*?)\\n\\]`, "m");
  const match = src.match(re);
  if (!match) throw new Error(`Cannot find ${varName} in source`);
  const body = match[1];

  // Each non-empty inner row: [ a, b, c, ... ],
  const rows = [];
  for (const line of body.split("\n")) {
    const rowMatch = line.match(/\[([^\]]+)\]/);
    if (!rowMatch) continue;
    const tokens = rowMatch[1].split(",").map((t) => t.trim()).filter(Boolean);
    const ints = tokens.map((t) => {
      if (t in symbols) return symbols[t];
      const n = Number(t);
      if (Number.isFinite(n)) return n;
      throw new Error(`Unknown token: ${t}`);
    });
    rows.push(ints);
  }
  if (rows.length === 0) throw new Error(`No rows parsed for ${varName}`);
  return rows;
}

function parsePalette(src, varName) {
  // Find: varName = { 1: (r, g, b), ... }
  const re = new RegExp(`${varName}\\s*=\\s*\\{([\\s\\S]*?)\\}`, "m");
  const match = src.match(re);
  if (!match) throw new Error(`Cannot find palette ${varName}`);
  const body = match[1];
  const palette = {};
  for (const line of body.split("\n")) {
    const m = line.match(/(\d+)\s*:\s*\(\s*(\d+)\s*,\s*(\d+)\s*,\s*(\d+)\s*\)/);
    if (m) palette[Number(m[1])] = [Number(m[2]), Number(m[3]), Number(m[4])];
  }
  return palette;
}

function emitTs(rows, palette) {
  return [
    `// AUTO-GENERATED by scripts/extract-sprites.mjs — do not edit by hand.`,
    `import type { Palette } from "../types.js";`,
    ``,
    `export const palette: Palette = ${JSON.stringify(palette, null, 2)};`,
    ``,
    `export const idle: number[][] = ${JSON.stringify(rows)};`,
    ``,
  ].join("\n");
}

for (const p of PROTOTYPES) {
  const src = readFileSync(p.pyPath, "utf8");
  const rows = parsePythonArray(src, p.arrayName, p.paletteSymbols);
  const palette = parsePalette(src, p.paletteName);
  mkdirSync(dirname(p.outPath), { recursive: true });
  writeFileSync(p.outPath, emitTs(rows, palette));
  console.log(`✓ ${p.outPath} (${rows.length} rows × ${rows[0].length} cols)`);
}
```

import { existsSync, readFileSync, readdirSync, statSync } from "node:fs";
import { join } from "node:path";
import type { MemoryBackend, MemoryContext } from "./backend.js";

export interface DefaultBackendOpts {
  cwd: string; // project root, where CLAUDE.md lives
  memoryDir?: string; // ~/.claude/projects/-/memory/ or similar
}

const SUMMARY_MAX = 800;

export class DefaultBackend implements MemoryBackend {
  constructor(private opts: DefaultBackendOpts) {}

  async fetchContext(): Promise<MemoryContext> {
    return {
      projectSummary: this.readClaudeMd(),
      userPreferences: this.readMemoryDir(),
    };
  }

  private readClaudeMd(): string {
    const path = join(this.opts.cwd, "CLAUDE.md");
    if (!existsSync(path)) return "";
    const raw = readFileSync(path, "utf8");
    return raw.length > SUMMARY_MAX ? raw.slice(0, SUMMARY_MAX) : raw;
  }

  private readMemoryDir(): string[] {
    const dir = this.opts.memoryDir;
    if (!dir || !existsSync(dir) || !statSync(dir).isDirectory()) return [];
    const files = readdirSync(dir).filter((f) => f.endsWith(".md"));
    return files.map((f) => readFileSync(join(dir, f), "utf8").trim());
  }
}

      petName: opts.petName,
      affection: opts.affection,

import { renderSprite } from "./halfblock.js";
import type { AnimatedSprite } from "../pets/types.js";

export interface PlayOpts {
  sprite: AnimatedSprite;
  finalFrameIndex: number;
  cycles: number;
  frameDelayMs: number;
  out?: NodeJS.WriteStream;
}

const sleep = (ms: number) => new Promise<void>((r) => setTimeout(r, ms));

export async function playAnimation(opts: PlayOpts): Promise<void> {
  const out = opts.out ?? process.stdout;
  const { sprite, cycles, frameDelayMs, finalFrameIndex } = sprite ? opts : opts;

  // Compute number of terminal lines per frame so we can move cursor up to redraw.
  const linesPerFrame = Math.ceil(sprite.frames[0].pixels.length / 2);
  let firstDraw = true;

  for (let c = 0; c < cycles; c++) {
    for (let f = 0; f < sprite.frames.length; f++) {
      if (!firstDraw) {
        // Move cursor up linesPerFrame and clear those lines
        out.write(`\x1b[${linesPerFrame}A`);
        out.write(`\x1b[J`);
      }
      out.write(renderSprite(sprite.frames[f]));
      firstDraw = false;
      await sleep(frameDelayMs);
    }
  }

  // Final settled frame
  out.write(`\x1b[${linesPerFrame}A`);
  out.write(`\x1b[J`);
  out.write(renderSprite(sprite.frames[finalFrameIndex]));
}

import type { Chattiness } from "../state/types.js";
import type { TemplateKey } from "../pets/types.js";

export type HookEvent = "SessionStart" | "Stop";

export interface DecideOpts {
  lastInteractionAt: string;
  chattiness: Chattiness;
  now: Date;
}

export function decideTrigger(event: HookEvent, opts: DecideOpts): TemplateKey | null {
  const last = new Date(opts.lastInteractionAt).getTime();
  const hoursAgo = (opts.now.getTime() - last) / 3_600_000;

  if (event === "SessionStart") {
    // Always speak on session start (it's a critical event)
    return hoursAgo >= 24 ? "session_return_long_absence" : "session_start_morning";
  }

  if (event === "Stop") {
    // Quiet skips non-critical Stop announcements
    if (opts.chattiness === "quiet") return null;
    return "task_completion";
  }

  return null;
}

## Out-of-MVP (defer to V1+)

The following design-doc features are explicitly **not** in the MVP and are tracked for a follow-up plan:

- `/feed`, `/play`, `/pet-stats`, `/pet-reset` commands
- Hook-driven proactive speech (SessionStart / Stop / error reactions)
- Chattiness levels and per-hour API call cap
- Full animation loops (currently only frame rotation in statusline)
- Premium pet pack distribution + activation key
- Multi-pet support
- Custom personality prompts (sponsor feature)
- Personal memory backend (separate private project)

Each becomes its own bite-sized plan after MVP ships and gets feedback.
